\begin{frame}
	\frametitle{NK model with long-run inflation expectations}
	\label{model_overview}
\begin{itemize}
\item New Keynesian core: standard IS and Phillips curves	%\hfill \hyperlink{HH}{\beamergotobutton{Microfoundations}}

 \begin{align*}
x_t &=  -\sigma i_t +\textcolor{brownreddark}{\hat{\E}_t} \sum_{T=t}^{\infty} \beta^{T-t }\big( (1-\beta)x_{T+1} - \sigma(\beta i_{T+1} - \pi_{T+1}) +\sigma r_T^n \big)   \\
\pi_t &= \kappa x_t +\textcolor{brownreddark}{\hat{\E}_t} \sum_{T=t}^{\infty} (\alpha\beta)^{T-t }\big( \kappa \alpha \beta x_{T+1} + (1-\alpha)\beta \pi_{T+1} + u_T\big)  
\end{align*}
%\hfill \hyperlink{derivations}{\beamergotobutton{Derivations}} \hyperlink{ALMs}{\beamergotobutton{Actual laws of motion}}

Observables: $(\pi, x, i)$ inflation, output gap, interest rate \\
Exogenous states: $(r^n, u)$ natural rate and cost-push shock
\
\pause
\item \textcolor{brownreddark}{Novelty of the paper}: inflation expectations process 
\begin{align*}
\textcolor{brownreddark}{\hat{\E}_{t}}\pi_{t+1} & = \textcolor{brownreddark}{\bar{\pi}_{t-1}} + \E_t\pi_{t+1}
%\bar{\pi}_{t}  &=\bar{\pi}_{t-1} +\mathbf{g}(f_{t|t-1}) \; f_{t|t-1}
\end{align*}

$\E$: rational (model-consistent) expectations \\
$\textcolor{brownreddark}{\hat{\E}}$: nonrational expectations $\rightarrow$ long-run inflation expectations $ \textcolor{brownreddark}{\bar{\pi}_{t-1}} $

\end{itemize}



\end{frame}

\begin{frame}
	\frametitle{Evolution of long-run inflation expectations}
	\label{adaptive_learning}


\begin{equation*}
\bar{\pi}_{t}  =\bar{\pi}_{t-1} +k_t  f_{t|t-1} 
\end{equation*}
 
 \

 \begin{itemize}
	 \item[] $ f_{t|t-1}  = \pi_{t} -  \hat{\E}_{t-1}\pi_{t} $
	 \item[] $ k_t \in [0,1]$ learning gain 
 \end{itemize} 
   
 

\

\

\pause

\textcolor{dodgerblue}{Endogenous gain} as metric for \textcolor{dodgerblue}{unanchoring}
\begin{equation*}
	k_{t}  =\mathbf{g}(f_{t|t-1}), \quad \quad \mathbf{g}'' > 0 % smooth, continuous, convex
\end{equation*}

\

\pause

\begin{itemize}
	\item If $k_t = 0 \rightarrow$ anchored expectations = rational expectations
	 \pause
	\item If $k_t = 1 \rightarrow$ short-run surprises translate 1:1 to long-run expectations
\end{itemize}

\end{frame}
